pcs_usage.c
#include <gmp.h>
#include <pcs.h>
#define MODULUS_SIZE 2048
int main(void)
{
hcs_rand *hr = hcs_rand_init();
pcs_public_key *pk = pcs_init_public_key();
pcs_private_key *vk = pcs_init_private_key();
pcs_generate_key_pair(pk, vk, hr, MODULUS_SIZE);
mpz_t a, b;
mpz_init_set_ui(a, 24125);
mpz_init_set_ui(b, 10);
pcs_encrypt(pk, hr, a, b);
pcs_ep_mul(pk, a, a, b);
pcs_ep_add(pk, a, a, b);
pcs_decrypt(vk, a, a);
gmp_printf("%Zd\n", a);
mpz_clear(a);
mpz_clear(b);
pcs_free_public_key(pk);
pcs_free_private_key(vk);
hcs_rand_free(hr);
}
